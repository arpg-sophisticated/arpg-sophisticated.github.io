68  def update_and_get_correction(self, error, delta_time):
69  self.integral += error * delta_time
70  if self.integral > self.anti_windup:
71  self.integral = self.anti_windup
72  elif self.integral < -self.anti_windup:
73  self.integral = -self.anti_windup
74 
75  derivative = (error - self.previous_error) / delta_time
76  self.previous_error = error
77  return self.p * error + self.i * self.integral + self.d * derivative
