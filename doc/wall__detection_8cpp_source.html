127  std::unordered_map<uint32_t, std::vector<pcl::PointXYZRGBL>*> mapClusters, std::vector<uint32_t> inputIgnoreIDs,
128  std::vector<pcl::PointXYZRGBL>* leftWall, std::vector<pcl::PointXYZRGBL>* rightWall)

134

136  {
137  uint32_t clusterID = cluster_wall.first;
138  if (std::find(inputIgnoreIDs.begin(), inputIgnoreIDs.end(), clusterID) != inputIgnoreIDs.end())
139  continue;
140 
141  std::vector<pcl::PointXYZRGBL>* cluster = cluster_wall.second;
142

144 
145  for (size_t i = 0; i < cluster->size(); i++)
146  {
147  Point p = { (*cluster)[i].x, (*cluster)[i].y };
148 
149  if (leftCircle.getDistance(p) < m_distance_threshold)
150  leftScore++;
151 
152  if (rightCircle.getDistance(p) < m_distance_threshold)
153  rightScore++;
154  }
155

159  {

162  additional_wall_ids.first.push_back(clusterID);
163  }
164  else if (rightScore > leftScore)

166  additional_wall_ids.second.push_back(clusterID);
167  }

257  std::unordered_map<uint32_t, std::vector<pcl::PointXYZRGBL>*> mapToCheck, float radius)
258 {
259  float maxLeft = 0;
260  float minRight = 0;
261  int64_t maxLeftID = -1;
262  int64_t minRightID = -1;
263 
264  for (auto itr = mapToCheck.begin(); itr != mapToCheck.end(); ++itr)
265  {
266  for (auto itrVector = itr->second->begin(); itrVector != itr->second->end(); ++itrVector)
267  {
268  double y = itrVector->y;
269  if (y < 0)
270  continue;
271  if ((itrVector->x > maxLeft) && (y <= radius))
272  {
273  maxLeft = itrVector->x;
274  maxLeftID = itrVector->label;
275  }
276  if ((itrVector->x < minRight) && (y <= radius))
277  {
278  minRight = itrVector->x;
279  minRightID = itrVector->label;
280  }
281  }
282  }
283 
284  if (maxLeftID == -1 && minRightID != -1)
285  {
286  // found a cluster for right but not for left. let's just choose the largest one for the left.
287  maxLeftID = findLargestCluster(mapToCheck, minRightID);
288  }
289  else if (minRightID == -1 && maxLeftID != -1)
290  {
291  // same but reverse
292  minRightID = findLargestCluster(mapToCheck, maxLeftID);
293  }
294 
295  return std::pair<int64_t, int64_t>(minRightID, maxLeftID);
296 }
